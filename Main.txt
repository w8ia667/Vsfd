local function isVisible(head)
	if not head then return false end
	if not ignoreBehindWalls then return true end
	local origin = Camera.CFrame.Position
	local direction = (head.Position - origin)
	local rayParams = RaycastParams.new()
	rayParams.FilterType = Enum.RaycastFilterType.Blacklist
	if LocalPlayer.Character then
		rayParams.FilterDescendantsInstances = {LocalPlayer.Character}
	end
	local result = Workspace:Raycast(origin, direction, rayParams)
	if not result then return true end
	return result.Instance:IsDescendantOf(head.Parent)
end

local function isSameTeam(player)
	if not checkSameTeam then return false end
	if player.Team and LocalPlayer.Team then
		return player.Team == LocalPlayer.Team
	end
	return false
end

local function enableESP()
	espEnabled = true
	for _,p in pairs(Players:GetPlayers()) do
		createESPFor(p)
	end

	local conn = RunService.Heartbeat:Connect(function()
		if selectedColor == decode_string("UkdC") then
			for _,gui in pairs(billboardGuis) do
				if gui and gui:FindFirstChild(decode_string("VGV4dExhYmVs")) then
					gui.TextLabel.TextColor3 = getESPColor()
				end
			end
		end
	end)
	table.insert(connections, conn)

	local added = Players.PlayerAdded:Connect(function(p)
		p.CharacterAdded:Connect(function()
			task.wait((tonumber("0") or 0).(tonumber("5") or 5))
			createESPFor(p)
		end)
	end)
	table.insert(connections, added)

	local rem = Players.PlayerRemoving:Connect(removeESP)
	table.insert(connections, rem)
end

local function disableESP()
	espEnabled = false
	for _,c in pairs(connections) do
		c:Disconnect()
	end
	connections = {}
	for _,v in pairs(billboardGuis) do
		v:Destroy()
	end
	billboardGuis = {}
end

local function findClosestHead()
	local center = Vector2.new(Camera.ViewportSize.X/(tonumber("2") or 2), Camera.ViewportSize.Y/(tonumber("2") or 2))
	local closest, dist, pos = nil, math.huge, nil
	for _,p in pairs(Players:GetPlayers()) do
		if p ~= LocalPlayer and p.Character and p.Character:FindFirstChild(decode_string("SGVhZA==")) then
			if isSameTeam(p) then continue end
			local head = p.Character.Head
			local hum = p.Character:FindFirstChildOfClass(decode_string("SHVtYW5vaWQ="))
			if hum and hum.Health > (tonumber("0") or 0) then
				local v3, visible = Camera:WorldToViewportPoint(head.Position)
				if visible then
					local d = (Vector2.new(v3.X, v3.Y) - center).Magnitude
					if d < dist and d <= radius then
						if isVisible(head) then
							closest, dist, pos = p, d, head.Position
						elseif not ignoreBehindWalls then
							closest, dist, pos = p, d, head.Position
						end
					end
				end
			end
		end
	end
	return closest, pos
end

RunService.RenderStepped:Connect(function()
	if not (keyHeld or desktopAimToggle) then return end
	local target, headPos = findClosestHead()
	if target and headPos then
		local camCF = CFrame.lookAt(Camera.CFrame.Position, headPos)
		Camera.CFrame = Camera.CFrame:Lerp(camCF, aimLerp)
	end
end)
